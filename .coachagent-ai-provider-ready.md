### 2. Update CoachAgent.ts

Replace the placeholder `handleChatMessage()` with actual AI integration:

```typescript
import { generateText } from 'ai';
import { openai } from '@ai-sdk/openai';
import { createCloudflare } from '@ai-sdk/cloudflare';

// In handleChatMessage():
private async handleChatMessage(
  connection: Connection,
  payload: { content: string; requestId?: string }
): Promise<void> {
  const { content, requestId } = payload;

  // Store user message
  this.sql`
    INSERT INTO conversation_history (role, content)
    VALUES ('user', ${content})
  `;

  // Get recent context
  const history = this.sql<ConversationMessage>`
    SELECT role, content FROM conversation_history
    ORDER BY created_at DESC LIMIT 10
  `.reverse();

  // Choose model based on available keys
  const model = this.env.OPENAI_API_KEY
    ? openai('gpt-4o-mini') // Cheaper, fast
    : createCloudflare({ binding: this.env.AI })('@cf/meta/llama-3.1-8b-instruct');

  // Build context with persona
  const systemPrompt = this.getPersonaPrompt();

  try {
    const { text } = await generateText({
      model,
      system: systemPrompt,
      messages: history.map(m => ({
        role: m.role as 'user' | 'assistant',
        content: m.content
      })),
      prompt: content,
    });

    // Store assistant response
    this.sql`
      INSERT INTO conversation_history (role, content)
      VALUES ('assistant', ${text})
    `;

    connection.send(JSON.stringify({
      type: 'chat_response',
      requestId,
      content: text,
      persona: this.state.personaMode,
    }));

  } catch (error) {
    console.error('[CoachAgent] AI error:', error);
    connection.send(JSON.stringify({
      type: 'error',
      code: 'AI_ERROR',
      message: 'Failed to generate response',
    }));
  }
}

// Add persona prompt builder
private getPersonaPrompt(): string {
  const prompts = {
    supportive: `You are a supportive, encouraging fitness coach. Celebrate wins, offer gentle guidance on setbacks, and keep the athlete motivated. Be warm and personable.`,
    analytical: `You are a data-driven performance coach. Focus on metrics, trends, and evidence-based recommendations. Be precise and thorough in your analysis.`,
    intense: `You are a no-nonsense, high-performance coach. Push the athlete hard, demand accountability, and expect maximum effort. Be direct and challenging.`,
    recovery: `You are a recovery and wellness focused coach. Prioritize rest, stress management, and sustainable training. Be calming and restorative.`
  };
  return prompts[this.state.personaMode];
}
```
